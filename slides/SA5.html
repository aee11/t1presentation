FLights.db Schema

CREATE TABLE Flights (
        flightNumber integer not null,
        airline text not null,
        departureTime text not null,
        arrivalTime text,
        departureLocation text,
        arrivalLocation text,
        price integer check (price >= 0),
        seatsAvailable integer check (seatsAvailable >= 0),
        primary key (flightNumber, airline, departureTime));
CREATE TABLE Passengers (
        bookingId integer references Bookings(bookingId),
        name text);
CREATE TABLE Bookings (
        bookingId integer primary key autoincrement,
        email text);
CREATE TABLE BookedFlights (
        bookingId integer references Bookings(bookingId),
        airline text,
        flightNumber integer,
        departureTime text,
        foreign key (airline, flightNumber, departureTime) 
        references Flights(airline, flightNumber, departureTime));
