Hotels.db Schema

CREATE TABLE "Hotels" (
	`hotelnames`	TEXT,
	`location`	TEXT,
	`rating`	INTEGER,
	`wifi`		TEXT,
	`roomPrice`	INTEGER
);
CREATE TABLE "Booking" (
	`name`		TEXT,
	`email`		TEXT,
	`phone`		TEXT,
	`numberofrooms`	INTEGER,
	`hotel`		TEXT,
	`totalprice`	INTEGER,
	`checkin`	TEXT,
	`checkout`	TEXT
);
CREATE TABLE "Available" (
		`hotel`		TEXT,
		`date`		TEXT,
		`roomsAvail`	INTEGER,
		PRIMARY KEY(hotel,date)
);
